************************************        # tuple
-----------------                           # создание
var tuple_0 : (String, Int)                 - вариант 0 - без параметров, без ключей
var tuple_1 = ("Sasha", 29)                 - вариант 1 - сразу с параметрами, но без ключей вызова параметров
var tuple_2 : (name: String, age: Int?)     - вариант 2 - без параметров, но с ключами
var tuple_3 = (name: "Sasha", age: 29)      - вариант 3 - сразу с параметрами и ключами вызова параметров (var tuple_3 = (name: var1, age: var2))
var (namez, agez) = tuple_2                 - вариант 4 - копирует полностью tuple_2, но со своими ключиками к вызову! (это отдельный тюпл!)

-----------------                           # изменения tuple
tuple_2.name = "xxx"                        - изменение по ключу
tuple_2.0    = "xxx"                        - изменение по индексу
namez        = "xxx"                        - изменение по новому ключу (скопированного tuple)

-----------------                           # вывод tuple
tuple_2.name
tuple_2.0
namez

-----------------                           # возможности с 2мя tuple's
let tuple_1 = (x:1, y:2)                    // создали tuple_1 с одинаковыми названиями как у tuple_2
var tuple_2 = (x:4, y:3)                    // создали tuple_2 с одинаковыми названиями как у tuple_1
tuple_2 = tuple_1                           // так можно (если в тюплз совпадают название и количество параметров!)

-----------------                           # примеры
var tuple_1 : (name: String, age: Int)      // создали tuple без параметров
tuple_1.name = "Sasha"                      // задали параметр имени
tuple_1.1 = 29                              // задали параметр возрасту

var (namez, agez) = tuple_1                 // копирует полностью tuple_2, но со своими ключиками к вызову!
namez = "Jesus"                             // изменил имя с "Sasha" на "Jesus"
agez = 33                                   // изменил возраст с 29 на 33

tuple_1.name                                // как был "Sasha" - так и остался
namez                                       // стал "Jesus"
